SELECT Είδος.Oid     
        ,Κωδικός
        ,Περιγραφή
        ,BarCode = null
        ,ΦΠΑ 
        ,ΜονάδεςΜέτρησης.ΜονάδαΜέτρησης
	 ,ΜονάδεςΜέτρησης.ΤύποςΔιάστασης
         ,Πλάτος
         ,Μήκος
         ,Υψος
         ,Κείμενο2 as ProductCode2
	 ,ΤιμήΧονδρικής
	 ,Χρώματα = null
	 ,Μεγέθη = null
FROM Είδος left join ΜονάδεςΜέτρησης on ΜονάδεςΜέτρησης.Oid = Είδος.ΜονάδαΜέτρησης
where ((Είδος.Κωδικός LIKE '%{0}%' OR  Είδος.Περιγραφή LIKE '%{0}%' or Είδος.Κείμενο2 LIKE '%{0}%') and (ΥποχρεωτικήΔιαχείρησηSN ='false' or ΥποχρεωτικήΔιαχείρησηSN is null)) and Είδος.GCRecord is null

UNION

SELECT Είδος as Oid
	,BarCode as Κωδικός
        ,Περιγραφή
        ,BarCode
	,ΦΠΑ
	,ΜονάδεςΜέτρησης.ΜονάδαΜέτρησης
	,ΜονάδεςΜέτρησης.ΤύποςΔιάστασης
	,Πλάτος
        ,Μήκος
        ,Υψος
	,Κείμενο2 as ProductCode2
        ,ΤιμήΧονδρικής
        ,Χρώματα.Χρώματα
        ,Μεγέθη.Μεγέθη
FROM BarCodeΕίδους
                                  left join Χρώματα on Χρώματα.Oid = BarCodeΕίδους.Χρώμα
                                  left join ΜονάδεςΜέτρησης on ΜονάδεςΜέτρησης.Oid = BarCodeΕίδους.ΜονάδαΜέτρησης
                                  left join Μεγέθη on Μεγέθη.Oid = BarCodeΕίδους.Μέγεθος
where  (BarCode LIKE '%{0}%' OR  Περιγραφή LIKE '%{0}%'  or BarCodeΕίδους.Κείμενο2 LIKE '%{0}%') and BarCodeΕίδους.GCRecord is null
